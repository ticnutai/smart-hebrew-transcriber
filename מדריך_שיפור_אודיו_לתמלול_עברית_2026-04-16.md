## 💻 דוגמאות קוד

### שלב 1: noisereduce

```python
import noisereduce as nr
import numpy as np
import soundfile as sf

# טעינת קובץ
audio, sr = sf.read("input.wav")

# הסרת רעש אוטומטית
cleaned = nr.reduce_noise(y=audio, sr=sr, prop_decrease=0.8)

# שמירה
sf.write("cleaned.wav", cleaned, sr)
```

### שלב 2: EQ + Normalize

```python
from scipy.signal import butter, sosfilt
from pydub import AudioSegment
from pydub.effects import normalize

# High-Pass Filter (הסרת תדרים מתחת ל-80Hz)
def highpass(audio, sr, cutoff=80):
    sos = butter(5, cutoff, btype='highpass', fs=sr, output='sos')
    return sosfilt(sos, audio)

# Low-Pass Filter (הסרת תדרים מעל 12KHz)
def lowpass(audio, sr, cutoff=12000):
    sos = butter(5, cutoff, btype='lowpass', fs=sr, output='sos')
    return sosfilt(sos, audio)

# נורמליזציה עם pydub
audio = AudioSegment.from_wav("cleaned.wav")
normalized = normalize(audio, headroom=1.0)
normalized.export("final.wav", format="wav")
```

### שלב 3: Demucs (אופציונלי)

import subprocess
